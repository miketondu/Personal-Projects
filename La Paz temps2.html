def leaflet_plot_stations(binsize, hashid):

    df = pd.read_csv('data/C2A2_data/BinSize_d{}.csv'.format(binsize))

    station_locations_by_hash = df[df['hash'] == hashid]

    lons = station_locations_by_hash['LONGITUDE'].tolist()
    lats = station_locations_by_hash['LATITUDE'].tolist()

    plt.figure(figsize=(8,8))

    plt.scatter(lons, lats, c='r', alpha=0.7, s=200)

    return mplleaflet.display()

df = pd.read_csv("data/C2A2_data/BinnedCsvs_d50/d058c00d7f0db633da3735126c04db328da4a3cf91deb92e28a0b527.csv")
df.set_index(pd.to_datetime(df['Date']),  inplace=True)

# Dividing data by .1 to turn units into understandable Celsius degrees
df.loc[:,'Data_Value'] *= 0.1

# Creating day, month, and year columns for easier plotting
df['Day'] = pd.DatetimeIndex(df['Date']).day 
df['Month'] = pd.DatetimeIndex(df['Date']).month 
df['Year'] = pd.DatetimeIndex(df['Date']).year 

df.set_index(['Month','Day'],inplace=True) 
df.sort_index(inplace = True)

# Dropping leap day February 29
selected_df = df.loc[2,29] 
df = df[~df.index.isin(selected_df.index)]
